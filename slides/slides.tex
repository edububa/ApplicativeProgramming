\begin{frame}{Applicative laws}
  \begin{itemize}
  \item Identity\\
    \verb~pure id <*> u = u~
  \item Composition\\
    \verb~pure (.) <*> u <*> v <*> w = u <*> (v <*> w)~
  \item Homomorphism\\
    \verb~pure f <*> pure x = pure (f x)~
  \item Interchange
    % \verb~u <*> pure x = pure (\f -> f x) <*> u~
  \end{itemize}
\end{frame}
